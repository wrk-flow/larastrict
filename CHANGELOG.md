  ```php  
      use MockeryPHPUnitIntegration;

      {  
      }  
  ```

  To run tests add this method when extending the TestCase.  
  ```php  
      /**  
       * @param \Closure(static):void $assert  
       * @dataProvider data  
       */  
      public function test(\Closure $assert): void  
      {  
          $assert($this);  
      }  
  ```
